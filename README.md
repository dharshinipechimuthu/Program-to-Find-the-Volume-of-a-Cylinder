# Program-to-Find-the-Volume-of-a-Cylinder
import math

# Input values
radius = 7
height = 10

# Calculate volume
volume = math.pi * radius * radius * height

# Display result
print("Radius of Cylinder :", radius)
print("Height of Cylinder :", height)
print("Volume of Cylinder :", round(volume, 2))

Output:
Radius of Cylinder : 7
Height of Cylinder : 10
Volume of Cylinder : 1539.38
